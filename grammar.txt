program      := block*

block        := interface
              | transform
              | pipeline
              | state
              | resource
              | event

interface    := "interface" NAME
                "inputs"  NAME*
                "outputs" NAME*

transform    := "transform" NAME
                "input"  NAME*
                "output" NAME*

pipeline     := "pipeline" NAME
                NAME+

state        := "state" NAME
                NAME+

resource     := "resource" NAME NAME

event        := "event" NAME
                NAME+
